### promise
##### 图片请求超时
```
let imgUrl = (url) =>{
    return new Promise((resolve,reject) => {
        let img = new Image();
        img.src = url;
        img.onload = () =>{
            console.log('图片请求成功');
            resolve(image);
        }
        img.oneerr = (err) => {
            console.log('图片请求失败');
            reject(err);
        }
    })
}
imgUrl(url).then(() => {
    console.log('success');
}).catch((error) => {
    console.log('error');
})
```
